        let isRunning = false;
        let shuffleInterval;
        const music = document.getElementById('bg-music');
        const synth = window.speechSynthesis;
        music.volume = 0.15;

        function initAudio() { music.play(); document.getElementById('btn-init').style.display = 'none'; }

        function toggleOracle() {
            initAudio();
            if (!isRunning) {
                isRunning = true;
                document.getElementById('scanner').style.display = 'block';
                document.getElementById('viewport').classList.add('shuffling');
                document.getElementById('challenge').style.display = 'none';
                document.getElementById('wisdom').innerText = "Vortex énergétique en cours...";
                
                shuffleInterval = setInterval(() => {
                    const rand = esprits[Math.floor(Math.random() * esprits.length)];
                    document.getElementById('spirit-img').src = rand.img;
                    document.getElementById('spirit-name').innerText = rand.nom;
                }, 60);
            } else {
                isRunning = false;
                clearInterval(shuffleInterval);
                document.getElementById('scanner').style.display = 'none';
                document.getElementById('viewport').classList.remove('shuffling');

                const final = esprits[Math.floor(Math.random() * esprits.length)];
                document.getElementById('spirit-img').src = final.img;
                document.getElementById('spirit-name').innerText = final.nom;
                document.getElementById('spirit-type').innerText = final.type;
                document.getElementById('wisdom').innerText = final.sagesse;
                document.getElementById('challenge').innerText = "DÉFI : " + final.defi;
                document.getElementById('challenge').style.display = 'block';

                speak(`${final.nom}. ${final.type}. ${final.sagesse}. Ton défi est le suivant : ${final.defi}`);
            }
        }

        function speak(text) {
            synth.cancel();
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.lang = 'fr-FR';
            utterance.volume = 1.0;
            utterance.rate = 0.95;
            synth.speak(utterance);
        }
